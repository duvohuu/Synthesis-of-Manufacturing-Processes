\chapter{TỔNG QUAN CÁC QUÁ TRÌNH CHẾ TẠO}
    \section{Phân loại vật liệu trong chế tạo}

        \begin{itemize}
            \item \textbf{Kim loại}
            \begin{itemize}
                \item Kim loại đen
                \begin{itemize}
                    \item Thép
                    \item Gang
                \end{itemize}
                \item Kim loại màu
                \begin{itemize}
                    \item Nhôm và hợp kim nhôm
                    \item Đồng và hợp kim đồng
                    \item Titan, niken, kẽm, thiếc, \dots
                \end{itemize}
            \end{itemize}

            \item \textbf{Gốm (Ceramics)}
            \begin{itemize}
                \item Gốm tinh thể
                \begin{itemize}
                    \item Gốm truyền thống (đất sét, sứ)
                    \item Gốm kỹ thuật (Al$_2$O$_3$, ZrO$_2$, \dots)
                \end{itemize}
                \item Thủy tinh
                \begin{itemize}
                    \item Chủ yếu dựa trên SiO$_2$
                \end{itemize}
            \end{itemize}

            \item \textbf{Polyme}
            \begin{itemize}
                \item Polyme nhiệt dẻo
                \item Polyme nhiệt rắn
                \item Elastomer (cao su)
            \end{itemize}

            \item \textbf{Composite}
            \begin{itemize}
                \item Composite cốt sợi
                \item Composite cốt hạt
            \end{itemize}
        \end{itemize}

    \section{Phân loại các quá trình chế tạo}

        \begin{itemize}
            \item \textbf{Quá trình gia công (Processing Operations)}
            \begin{itemize}
                \item \textbf{Quá trình tạo hình (Shaping Processes)}
                \begin{itemize}
                    \item Quá trình đông đặc (đúc)
                    \item Gia công biến dạng (rèn, cán, ép đùn, kéo)
                    \item Gia công bột (ép bột, thiêu kết)
                    \item Gia công cắt gọt (tiện, phay, khoan, mài)
                \end{itemize}

                \item \textbf{Quá trình cải thiện tính chất}
                \begin{itemize}
                    \item Nhiệt luyện kim loại
                    \item Thiêu kết vật liệu bột và gốm
                \end{itemize}

                \item \textbf{Quá trình xử lý bề mặt}
                \begin{itemize}
                    \item Làm sạch bề mặt
                    \item Xử lý bề mặt cơ – lý – hóa
                    \item Phủ và mạ bề mặt
                \end{itemize}
            \end{itemize}

            \item \textbf{Quá trình lắp ráp (Assembly Operations)}
            \begin{itemize}
                \item Liên kết vĩnh viễn
                \begin{itemize}
                    \item Hàn nóng chảy
                    \item Hàn thau
                    \item Hàn mềm
                    \item Dán keo
                \end{itemize}
                \item Lắp ghép cơ khí
                \begin{itemize}
                    \item Bu lông, đai ốc
                    \item Lắp ép, lắp chặt
                \end{itemize}
            \end{itemize}
        \end{itemize}
